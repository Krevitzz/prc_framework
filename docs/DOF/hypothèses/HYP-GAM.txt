\HYPOTHESIS{
  ID = HYP-GAM-001
  LEVEL = L2
  DEPENDENCIES = {
    axioms = [A2, A3]
    definitions = [DEF-GAMMA]
    rejected_hypotheses = []
  }
  STATEMENT = {
    Γ peut être défini comme une transformation non-linéaire 
    de saturation appliquée uniformément à C :
    
    C_{t+1} = tanh(β · C_t)
    
    où β contrôle la vitesse de saturation.
    
    Cette forme préserve la structure (signe des corrélations)
    tout en limitant leur croissance, sans mécanisme 
    de renforcement additionnel.
  }
  STATUS = WIP
  NOTES = {
    - Hypothèse minimale : régulation seule, sans transitivité
    - Teste si la limitation de croissance suffit à stabiliser D
    - Compatible A3 : ne cherche pas à annuler D, juste à le borner
    - Plus simple que Hebbian + Saturation (HYP-GAM-002 potentielle)
  }
}
